   /**

   public enum StationType {

     cutting_board(3),
     sink(4),
     bin(5),
     ingredient(6),
     serve(7);
 
     private final int value;

       this.value = id;
     }

      * Get the id of this station type.

      * @return integer id.

     public int getValue() {

     }
 
     private static final Map<Integer, StationType> _map = new HashMap<>();

       for (StationType stationType : StationType.values()) {

       }
     }
